user (id, username, email, password)

bahan (id, nama, jumlah, kedaluwarsa)
resep (id_coklat, bahan, jumlah, harga)  *tipe bahan dan jumlah adalah array of int
add_stock (id, id_coklat, jumlah, status)
saldo (id, jumlah)

create table user (
    id int primary key auto_increment,
    username varchar(255) not null,
    password varchar(255) not null
);

create table coklat (
    id int primary key auto_increment,
    nama varchar(256) not null,
    harga int not null,
    jumlah int not null,
    deskripsi text
);

create table bahan (
    id int primary key auto_increment,
    nama varchar(255) not null,
    jumlah int not null,
    kedaluarsa timestamp
);
